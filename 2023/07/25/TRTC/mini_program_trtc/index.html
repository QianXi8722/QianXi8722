小程序–实现实时音视频–进入房间后的操作步骤

实时音视频流程

  1. 导入SDK
  2. 配置app权限
  3. 进入房间（当前文章从此步骤开始讲）
  4. 订阅音视频流
  5. 发布音视频流
  6. 退出房间

步骤一：进入房间

// 事件监听
bindTRTCRoomEvent() {
    const TRTC_EVENT = this.TRTC.EVENT
    // 初始化事件订阅
    this.TRTC.on(TRTC_EVENT.LOCAL_JOIN, (event) => {
        console.log('* room LOCAL_JOIN', event)
    })
    this.TRTC.on(TRTC_EVENT.LOCAL_LEAVE, (event) => {
        console.log('* room LOCAL_LEAVE', event)
    })
    this.TRTC.on(TRTC_EVENT.ERROR, (event) => {
        console.log('* room ERROR', event)
    })
    // 远端用户加入
    this.TRTC.on(TRTC_EVENT.REMOTE_USER_JOIN, (event) => {
        console.error('* room REMOTE_USER_JOIN', event)
    })
    // 远端用户退出
    this.TRTC.on(TRTC_EVENT.REMOTE_USER_LEAVE, (event) => {
        console.error('* room REMOTE_USER_LEAVE', event)
    })
    // 远端用户推送视频
    this.TRTC.on(TRTC_EVENT.REMOTE_VIDEO_ADD, (event) => {
        console.log('* room REMOTE_VIDEO_ADD',  event)
    })
    // 远端用户取消推送视频
    this.TRTC.on(TRTC_EVENT.REMOTE_VIDEO_REMOVE, (event) => {
        console.log('* room REMOTE_VIDEO_REMOVE', event)
    })
    // 远端用户推送音频
    this.TRTC.on(TRTC_EVENT.REMOTE_AUDIO_ADD, (event) => {
        console.log('* room REMOTE_AUDIO_ADD', event)
    })
    // 远端用户取消推送音频
    this.TRTC.on(TRTC_EVENT.REMOTE_AUDIO_REMOVE, (event) => {
        console.log('* room REMOTE_AUDIO_REMOVE', event)
    })
    // 远端用户音量更新
    this.TRTC.on(TRTC_EVENT.REMOTE_AUDIO_VOLUME_UPDATE, (event) => {
        console.log('* room REMOTE_AUDIO_VOLUME_UPDATE', event)
    })
    // 本地用户音量更新
    this.TRTC.on(TRTC_EVENT.LOCAL_AUDIO_VOLUME_UPDATE, (event) => {
        console.log('* room LOCAL_AUDIO_VOLUME_UPDATE', event)
    })
},

步骤二：订阅音视频流

// 远端用户推送视频
this.TRTC.on(TRTC_EVENT.REMOTE_VIDEO_ADD, (event) => {
  console.log('* room REMOTE_VIDEO_ADD',  event)
  const { player } = event.data
  // 开始播放远端的视频流，默认是不播放的
  this.setPlayerAttributesHandler(player, { muteVideo: false })
})
// 远端用户推送音频
this.TRTC.on(TRTC_EVENT.REMOTE_AUDIO_ADD, (event) => {

  const { player } = event.data
  this.setPlayerAttributesHandler(player, { muteAudio: false })
})
// 设置某个 player 属性

  this.setData({

步骤四：退出房间

  • 主动退出当前房间
    exitRoom() {
    const result = this.TRTC.exitRoom()
    // 状态机重置，会返回更新后的pusher和playerList
    this.setData({
        pusher: result.pusher,
        playerList: result.playerList,
    })
},
  • 被动退出当前房间
    let onKickedout = function(event){
  console.log('被服务端踢出或房间被解散')
}
this.TRTC.on(EVENT.KICKED_OUT, onKickedout)

    解散房间和退出房间不一样
    解散房间：是由房主 主动解散房间
    退出房间：是成员退出房间会监听，房间仍然存在。和进入房间一样

示例

/**
   * 页面的初始数据
   */
  data: {
    _rtcConfig: {
      sdkAppID: '', // 必要参数 开通实时音视频服务创建应用后分配的 sdkAppID
      userID: '', // 必要参数 用户 ID 可以由您的帐号系统指定
      userSig: '', // 必要参数 身份签名，相当于登录密码的作用
    },
    modelTip: config.modelTip, // 弹框标题
    pusher: null, // live-pusher 参数
    playerList: [], // live-player
    localAudio: true, // 进入房间时是否打开音频
    localVideo: true, // 进入房间时是否打开摄像头

    imageUrl: config.hs_resource_url + "/static/future_pharmacy/", // 图片地址
    cameraModalShow: false, // 打开麦克风摄像头弹框
    // quitModalShow: false, // 退出确认弹框
    modalBtn: [{
        label: '取消',
        value: 0
      },
      {
        label: '确定',
        value: 1,
        color: '#1E53F9'
      }
    ],
  },

  /**
   * 生命周期函数--监听页面加载
   */
  onLoad(options) {
    this.setData({
      cameraModalShow: true
    });
    console.log('room onload', options);
    // 保持屏幕常亮
    wx.setKeepScreenOn({
      keepScreenOn: true,
    });
    // trtc实例
    this.TRTC = new TRTC(this);
    // 将String 类型的 true false 转换成 boolean
    Object.getOwnPropertyNames(options).forEach((key) => {
      if (options[key] === 'true') {
        options[key] = true
      }
      if (options[key] === 'false') {
        options[key] = false
      }
    });
    // 初始化 pusher 参数
    this.init(options);
    // 事件监听
    this.bindTRTCRoomEvent();
    this.enterRoom({
      roomID: options.roomID
    });
  },
  /**
   * 生命周期函数--监听页面销毁
   */
  onUnload() {
    if (app.globalData.queueStatus == 2) {
      if (app.globalData.orderConfirmData.source_form == 4) {
        console.log('取消药学服务');
        // 取消药学服务
        this.cancelMedicalConsultation();
      } else {
        // 取消问诊
        this.cancelConsultation();
      }
    }
  },

  init(options) {
    // pusher 初始化参数
    const pusherConfig = {
      beautyLevel: 0
    }
    const pusher = this.TRTC.createPusher(pusherConfig);
    this.setData({
      _rtcConfig: {
        userID: options.userID,
        sdkAppID: options.sdkAppID,
        userSig: options.userSig,
      },
      pusher: pusher.pusherAttributes,
      localAudio: options.localAudio,
      localVideo: options.localVideo
    })
  },

  // 进入房间
  enterRoom(options) {
    const roomID = options.roomID;
    const config = Object.assign(this.data._rtcConfig, {
      roomID
    });
    this.setData({
      pusher: this.TRTC.enterRoom(config),
    }, () => {
      this.TRTC.getPusherInstance().start(); // 开始推流
    })
  },

  // 退出房间
  exitRoom() {
    const result = this.TRTC.exitRoom();
    this.setData({
      pusher: result.pusher,
      playerList: result.playerList,
    })
  },

  // 设置 pusher 属性
  setPusherAttributesHandler(options) {
    this.setData({
      pusher: this.TRTC.setPusherAttributes(options),
    })
  },

  // 设置某个 player 属性
  setPlayerAttributesHandler(player, options) {
    this.setData({
      playerList: this.TRTC.setPlayerAttributes(player.streamID, options),
    })
  },

  // 事件监听
  bindTRTCRoomEvent() {
    const TRTC_EVENT = this.TRTC.EVENT;
    // 本地进房成功后的回调
    this.TRTC.on(TRTC_EVENT.LOCAL_JOIN, (event) => {
      console.debug('* room LOCAL_JOIN', event);
      if (this.data.localVideo) {
        this.setPusherAttributesHandler({
          enableCamera: true
        });
      }
      if (this.data.localAudio) {
        this.setPusherAttributesHandler({
          enableMic: true
        });
      }
    });
    // 本地离开房间后的回调
    this.TRTC.on(TRTC_EVENT.LOCAL_LEAVE, (event) => {
      console.debug('* room LOCAL_LEAVE', event);
    })
    this.TRTC.on(TRTC_EVENT.ERROR, (event) => {
      console.debug('* room ERROR', event);
    })
    // 解散房间
    this.TRTC.on(TRTC_EVENT.KICKED_OUT, event => {
      console.debug('房间被解散1', event);
      if (app.globalData.orderConfirmData.source_form == 4) {
        console.log('source_form === 4');
        // 取消药学服务
        // this.cancelMedicalConsultation();
        // 跳转确认订单页面
        this.pharmaceuticalCare()
      }
    })
    // 远端用户加入此房间
    this.TRTC.on(TRTC_EVENT.REMOTE_USER_JOIN, (event) => {
      console.debug('* room REMOTE_USER_JOIN', event);
      const {
        playerList
      } = event.data;
      this.setData({
        playerList: playerList
      })
    })
    // 远端用户离开此房间
    this.TRTC.on(TRTC_EVENT.REMOTE_USER_LEAVE, (event) => {
      console.debug('* room REMOTE_USER_LEAVE', event);
      const {
        playerList
      } = event.data;
      this.setData({
        playerList: playerList
      })
    })
    // 远端用户推送视频
    this.TRTC.on(TRTC_EVENT.REMOTE_VIDEO_ADD, (event) => {
      console.debug('* room REMOTE_VIDEO_ADD', event);
      const {
        player
      } = event.data;
      // 开始播放远端的视频流，默认是不播放的
      this.setPlayerAttributesHandler(player, {
        muteVideo: false
      });
    })
    // 远端用户取消推送视频
    this.TRTC.on(TRTC_EVENT.REMOTE_VIDEO_REMOVE, (event) => {
      console.debug('* room REMOTE_VIDEO_REMOVE', event);
      const {
        player
      } = event.data;
      this.setPlayerAttributesHandler(player, {
        muteVideo: true
      });
    })
    // 远端用户推送音频
    this.TRTC.on(TRTC_EVENT.REMOTE_AUDIO_ADD, (event) => {
      console.debug('* room REMOTE_AUDIO_ADD', event);
      const {
        player
      } = event.data;
      this.setPlayerAttributesHandler(player, {
        muteAudio: false
      });
    })
    // 远端用户取消推送音频
    this.TRTC.on(TRTC_EVENT.REMOTE_AUDIO_REMOVE, (event) => {
      console.debug('* room REMOTE_AUDIO_REMOVE', event);
      const {
        player
      } = event.data;
      this.setPlayerAttributesHandler(player, {
        muteAudio: true
      });
    })
    this.TRTC.on(TRTC_EVENT.REMOTE_AUDIO_VOLUME_UPDATE, (event) => {
      const {
        playerList
      } = event.data;
      this.setData({
        playerList: playerList
      })
    })
    this.TRTC.on(TRTC_EVENT.LOCAL_AUDIO_VOLUME_UPDATE, (event) => {
      const {
        pusher
      } = event.data;
      this.setData({
        pusher: pusher
      })
    })
  },

  // 设置美颜
  _setPusherBeautyHandle() {
    const beautyLevel = this.data.pusher.beautyLevel === 0 ? 9 : 0;
    this.setPusherAttributesHandler({
      beautyLevel
    });
  },

  // 开关音频
  _pusherAudioHandler() {
    if (this.data.pusher.enableMic) {
      this.setPusherAttributesHandler({
        enableMic: false
      });
    } else {
      this.setPusherAttributesHandler({
        enableMic: true
      });
    }
  },

  // 开关摄像头
  _pusherVideoHandler() {
    if (this.data.pusher.enableCamera) {
      this.setPusherAttributesHandler({
        enableCamera: false
      });
    } else {
      this.setPusherAttributesHandler({
        enableCamera: true
      });
    }
  },

  // 状态变化事件

  /**
   * 打开麦克风摄像头弹框按钮点击

  handleCameraModalTap(e) {
    this.setData({
      cameraModalShow: false
    });

      this._pusherAudioHandler();
      this._pusherVideoHandler();
    }
  },
  // 挂断退出房间
  async _hangUp() {
    console.log(app.globalData.orderConfirmData.source_form)
    try {
      // 取消排队
      app.globalData.queueStatus = 0;
      // 药学服务挂断
      if (app.globalData.orderConfirmData.source_form == 4) {
        // 取消药学服务咨询
        await this.cancelMedicalConsultation();
        // 跳转确认订单页面
        this.pharmaceuticalCare();
      } else {
        // 其他挂断，返回首页
        util.showToast('您已结束问诊!');
        this.exitRoom();
        setTimeout(() => {
          route.setRoute('switchTab', '/pages/tabBar/index/index');
        }, 1500)
      }
    } catch (error) {
      console.log(error);
    }
  },
  // 药学服务跳转确认订单页面
  pharmaceuticalCare() {

      company_id: app.globalData.orderConfirmData.company_id, // 门店id
      commodityInfo: app.globalData.orderConfirmData.commodityInfo,
      deviceInfo: app.globalData.orderConfirmData.deviceInfo,
      prescDrug: 3
    }
    app.globalData.orderConfirmData = orderConfirmData;
    // setTimeout(() => {
    //   route.setRoute('redirectTo', '/pages/medicine/view/drugOrderConfirm/drugOrderConfirm');
    // }, 1500);
    this.pageRouter.navigateBack();
  },
  /**
   * 取消问诊
   */
  async cancelConsultation() {
    try {
      await service.inquiry.cancelConsult({
        treatment_id: app.globalData.orderConfirmData.source_form === 1 ? app.globalData.orderConfirmData.orderInfo
          .treatment_id : '',
        room_id: this.data._rtcConfig.userID,
        doctor_id: app.globalData.orderConfirmData.doctorInfo.id,
        user_id: this.data._rtcConfig.userID
      });
    } catch (error) {
      console.log(error);
    }
  },
  // 取消药学服务
  async cancelMedicalConsultation() {
    try {
      let cancelServiceQuery = app.globalData.cancelServiceQuery;
      console.log('取消药学服务参数cancelServiceQuery', cancelServiceQuery);
      // 取消甲类OTC咨询
      let res = await service.confirmOrder.cancelConsultYx(cancelServiceQuery);
      console.log('取消药学服务res', res);
    } catch (error) {
      console.log(error);
    }
  },
